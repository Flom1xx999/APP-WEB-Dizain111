1. Тема та мета

Тема: Функціональне застосування JavaScript у HTML-документі.

Мета: ознайомитись з підключенням JavaScript до сторінки, обробкою подій та роботою зі змінними й масивами.

2. Підключення JavaScript

<!-- варіант 1: внутрішній скрипт -->
<script>
  console.log("Hello from JS");
</script>

<!-- варіант 2: зовнішній файл -->
<script src="script.js"></script>

3. Приклад взаємодії з користувачем

<button id="btn-hello">Привітати користувача</button>
<p id="hello-output"></p>

<script>
  const btn = document.getElementById("btn-hello");
  const out = document.getElementById("hello-output");

  btn.addEventListener("click", function () {
    const name = prompt("Введіть ваше імʼя:");
    if (name) {
      out.textContent = "Привіт, " + name + "!";
    }
  });
</script>

4. Приклад роботи з масивом цін

<button id="btn-total">Порахувати суму замовлення</button>
<p id="total-output"></p>

<script>
  const prices = [20000, 15000, 5000]; // приклад цін товарів у кошику
  const btnTotal = document.getElementById("btn-total");
  const outTotal = document.getElementById("total-output");

  btnTotal.addEventListener("click", function () {
    let sum = 0;
    for (let i = 0; i < prices.length; i++) {
      sum += prices[i];
    }
    outTotal.textContent = "Сума замовлення: " + sum + " грн";
  });
</script>

5. Перевірка введених даних форми

<form id="order-form">
  <label>Телефон:
    <input type="text" id="phone">
  </label>
  <button type="submit">Відправити</button>
</form>
<p id="form-message"></p>

<script>
  const form = document.getElementById("order-form");
  const phone = document.getElementById("phone");
  const msg = document.getElementById("form-message");

  form.addEventListener("submit", function (e) {
    e.preventDefault();
    if (phone.value.trim().length < 10) {
      msg.textContent = "Помилка: введіть коректний номер телефону.";
      msg.style.color = "tomato";
    } else {
      msg.textContent = "Дякуємо! Замовлення відправлено.";
      msg.style.color = "lightgreen";
    }
  });
</script>
